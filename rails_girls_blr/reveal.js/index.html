GET

Let us get Twitter's public timeline 

                    
                require 'json'
                require 'httparty'

                include HTTParty


                response =  HTTParty.get("http://rubygems.org/api/v1/versions/httparty.json")

                response.first.each do |key,value|
                    puts "#{key}:#{value}"
                end
